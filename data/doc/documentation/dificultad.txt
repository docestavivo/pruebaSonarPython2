#DIFICULTAD

LA DIFICULTAD SE DEFINE EN:
[ATK_MOB,DEF_MOB,ATK_PLAYER,DEF_PLAYER,COIN,TIME_MOVE_MOBS,TIME_POWERS,FOLLOWERS]

ATK_MOB : LA DIFICULTAD DEL ATAQUE DEL MOB SUBIRA EN FUNCION DE LA DIFICULTAD
DEF_MOB : LA DEFENSA DEL ATAQUE DEL MOB SUBIRA EN FUNCION DE LA DIFICULTAD
ATK_PLAYER: EL ATAQUE DEL JUGADOR DISMINUIRA EN FUNCION DE LA DIFICULTAD
DEF_PLAYER: LA DEFENSA DEL JUGADOR DISMINUIRA EN FUNCION DE LA DIFICULTAD
COIN: EL DINERO GANADO DISMINUIRA EN FUNCION DE LA DIFICULTAD
TIME_MOVE_MOBS: TIEMPO EN QUE LOS MOBS SE MOVERAN
TIME_POWERS: TIEMPO DE CADA PODER
FOLLOWERS: SE DEFINE LA RAZON PARA CALCULAR EL ATK, HP Y DEF DE LOS FOLLOWER

#TODOS LOS NUMEROS IRAN ENTRE 0 Y 1

LOS PROMEDIOS SE CALCULAN COMO:
ATK (MOB) = NORMAL_ATK*(ATK_MOB+1)
DEF (MOB) = NORMAL_DEF*(DEF_MOB+1)
ATK (PLAYER) = NORMAL_ATK*(ATK_PLAYER+1)
DEF (PLAYER) = NORMAL_DEF*(DEF_PLAYER+1)
COIN = COIN*(COIN+1)
TIME = TIME_MOVE_MOBS_NORMAL*TIME_MOVE_MOBS
TIME_POWER = TIME*(1-TIME)
